 *Purpose: count.c under normal operations reads a single positve integer (N)  followed by any number of whitespace or new
 *lines, with N number of integers that follow. Count will then return the number of times each unique integer was seen in
 *the input stream. This occures after all integers are read. If less then N integers are read, program will continue to
 *read from io until the Nth integer ir read in, at which point it will output. If more then N integers are in stream, only
 *N integers will be evaluated. Uses mallac with (N * sizeof(int)) as size.
 *Error(s): N is not positve - fatal error msg is output to stderr return (1).
 */

#include <stdlib.h>
#include <stdio.h>
#include <ctype.h>

void bubbleSort(int*,int*,int);
void swap(int*,int*);

int main(void){
        int N,i,j,uniqueCount = 0;
        int ret;

        scanf("%d",&N);

        int *intArray = (int *)malloc(N * sizeof(int));

        if (intArray == NULL){
                fprintf(stderr,"First integer was not positve");
                return 1;
        }

        while(i < N) { //ret = scanf("%d", &value) != EOF was other condition in while but removed

                ret = scanf("%d", &intArray[i]);
                if (ret > 0) {
                        i++;
                }
                else {
                        fprintf(stderr, "input was not %d integers.", N);
                        return 1;
                }
        }

        int *uniqueInts = (int *)malloc(N * sizeof(int));
        int *counts = (int *)malloc(N * sizeof(int));

        for (i = 0; i < N; i++){
                int found = 0;

                for (j = 0; j < uniqueCount; j++){
                        if(intArray[i] == uniqueInts[j]){
                                counts[j]++;
                                found = 1;
                                break;
                        }

                }

                if(found == 0){
                        uniqueInts[uniqueCount] = intArray[i];
                        counts[uniqueCount] = 1;
                        uniqueCount++;
                }
        }

        bubbleSort(uniqueInts, counts, uniqueCount);

        for (i = 0; i < uniqueCount; i++){
                printf("%d %d\n", uniqueInts[i], counts[i]);
        }

        free(intArray);
        free(uniqueInts);
        free(counts);

        return 0;
}

//standard bubble sort, but works on two arrays simotaniously to keep integer and count together.
void bubbleSort(int *uniqueInts, int *counts, int uniqueCount){
        int i,j;
        for (i = 0; i < uniqueCount - 1; i++){
                for (j = 0; j < uniqueCount - i - 1; j++){
                        if (uniqueInts[j] > uniqueInts[j + 1]){
                                        swap(&uniqueInts[j], &uniqueInts[j + 1]);
                                        swap(&counts[j], &counts[j + 1]);
                        }
                }
        }
}

//stand triangle assignment for swap
void swap(int *a, int *b){
        int temp = *a;
        *a = *b;
        *b = temp;
}
