#include <stdio.h>
#include <stdlib.h>


// Function to swap two integers
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

// Function to sort uniqueNumbers and counts arrays in ascending order based on uniqueNumbers
void sortArrays(int *uniqueNumbers, int *counts, int uniqueCount) {
    int i, j;
    for (i = 0; i < uniqueCount - 1; i++) {
        for (j = 0; j < uniqueCount - i - 1; j++) {
            if (uniqueNumbers[j] > uniqueNumbers[j + 1]) {
                // Swap the numbers
                swap(&uniqueNumbers[j], &uniqueNumbers[j + 1]);
                // Swap the corresponding counts
                swap(&counts[j], &counts[j + 1]);
            }
        }
    }
}

int main() {
    int n, i, j, uniqueCount = 0;

    // Read the number of integers
    printf("Enter the number of integers: ");
    scanf("%d", &n);

    // Allocate memory to store the sequence of integers
    int *arr = (int *)malloc(n * sizeof(int));
    if (arr == NULL) {
        printf("Memory allocation failed\n");
        return 1;
    }

    // Read the integers from input
    printf("Enter the integers:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Allocate memory for storing the counts of unique numbers
    int *uniqueNumbers = (int *)malloc(n * sizeof(int));
    int *counts = (int *)malloc(n * sizeof(int));

    if (uniqueNumbers == NULL || counts == NULL) {
        printf("Memory allocation failed\n");
        free(arr);
        return 1;
    }

    // Count occurrences of each integer
    for (i = 0; i < n; i++) {
        int found = 0;
        // Check if the number is already counted
        for (j = 0; j < uniqueCount; j++) {
            if (arr[i] == uniqueNumbers[j]) {
                counts[j]++;
                found = 1;
                break;
            }
        }
        // If the number is not found, add it to uniqueNumbers
        if (!found) {
            uniqueNumbers[uniqueCount] = arr[i];
            counts[uniqueCount] = 1;
            uniqueCount++;
        }
    }

    // Sort the uniqueNumbers and their corresponding counts in ascending order
    sortArrays(uniqueNumbers, counts, uniqueCount);

    // Output the results
    printf("Number counts (sorted in ascending order):\n");
    for (i = 0; i < uniqueCount; i++) {
        printf("%d appears %d times\n", uniqueNumbers[i], counts[i]);
    }

    // Free allocated memory
    free(arr);
    free(uniqueNumbers);
    free(counts);

    return 0;
}
