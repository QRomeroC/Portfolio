/* File: anagrams.c
 * Author: Eric Anson
 * Purpose: This program reads in a series of strings and
 * outputs the strings that
 * are anagrams of the first string
 */
#include <stdio.h>
#include <ctype.h>

/* processFirst(str, cnt) takes a string (str) and an array of 26 ints cnt
 * It counts the letters in str, and puts the appropriate value in cnt.
 * It returns 1 if successful, or 0 if a nonletter is encountered.
 */
int processFirst (char *str, int *cnts) {
  int i;
  for (i=0; i < 26; ++i)
    cnts[i] = 0;
  while (*str != '\0') {
    if (!isalpha(*str))
      return 0;
    cnts[toupper(*str)-'A'] += 1;
    ++str;
  }
  return 1;
}

/* isAnnagram(str, cnts) takes a str and an integer array of 26.
 * It returns -1 if str contains non-alphabetic characters,
 *             1 if str is an anagram (has same char count as cnts)
 *             0 otherwise
 */
int isAnagram (char *str, int *cnts) {
  int newCnt[26];
  int i;
  for (i=0; i < 26; ++i)
    newCnt[i] = 0;
  while (*str != '\0') {
    if (!isalpha(*str))
      return -1;                //error in string
    newCnt[toupper(*str)-'A'] += 1;
    ++str;
  }
  for (i=0; i < 26; ++i)
    if (cnts[i] != newCnt[i])
      return 0;
  return 1;
}

/* main() - The driver program which reads in the strings.
 * It sends the first string to be processed and then calls a proc to
 * check if the other strings are anagrams. It assumes all strings are
 * 64 chars or less.
 */
int main () {
  char str[65];
  int counts[26];
  int errStat;
  int isAn;

  errStat = 0;          //no errors yet

  if (scanf("%64s", str) == EOF)
    return 0;    //no error, but we're done
  if (!processFirst(str, counts)) {
    fprintf(stderr,"Bad first string\n");
    return 1;
  }
  printf("%s\n",str);   //First string is always anagram

  while (scanf("%64s",str) != EOF) {
    isAn = isAnagram(str, counts);
    if (isAn < 0) {   //isAnagram return -1 if there was an error
      fprintf(stderr,"Bad Input ... non-alphabetical character\n");
      errStat = 1;
    }
    else if (isAn)
      printf("%s\n",str);
  }
  return errStat;

  return 0;
}
