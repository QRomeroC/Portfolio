 *Purpose: Reads from input a sequence of whitespace/newline
 *seperated strings and stores them in a linked list made
 *of nodes that have additional refrances to links of
 *anagrams of the root node. All non anagrams strins are
 *pointed to by the main link referances.
 *Error(s):
 *
 */

//header files
#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <string.h>
//stuct declartion
typedef struct strNode{
        char *word;
        struct strNode *groupNext;
}strNode;

typedef struct countNode{
        int count[26];
        struct countNode *nextCount;
        strNode *strHd,*strTail;
}countNode;
//prototypes
int badWord();
void initCountArray();
void updateCountArray(char*);
void updateCountNode(countNode*);
int checkCounts(countNode*);
void printList();
//globals
char currWord[65];
int countArray[26];
countNode *countHd,*countTail;

int main(void){
        int scanVal;
        int errSeen = 0;
        scanVal = scanf("%64s", currWord);

        while (badWord()){
                errSeen = 1;
                scanVal = scanf("%64s", currWord);
        }

        strNode *newStr = calloc(1,sizeof(strNode));
        if (newStr == NULL){
                fprintf(stderr,"Memory Allocation failure.\n");
                return 1;
        }

        newStr->word = strdup(currWord);
        if(newStr->word == NULL){
                fprintf(stderr,"Memory Allocation Failure.\n");
                return 1;
        }

        countNode *newCount = calloc(1,sizeof(countNode));
        if (newCount == NULL){
                fprintf(stderr,"Memory Allocation failure.\n");
                return 1;
        }

        initCountArray();
        updateCountArray(currWord);
        updateCountNode(newCount);

        newCount->strHd = newCount->strTail = newStr;
        newCount->nextCount = NULL;
        countHd = countTail = newCount;

        while ((scanVal = scanf("%64s", currWord)) != EOF){
                if(badWord()){
                        errSeen = 1;
                        break;
                }

                strNode *newStr = calloc(1,sizeof(strNode));
                if(newStr == NULL){
                        fprintf(stderr,"Memory Allocation failure.\n");
                        return 1;
                }
                newStr->word = strdup(currWord);
                if(newStr->word == NULL){
                        fprintf(stderr,"Memory Allocation failure.\n");
                        return 1;
                }

                initCountArray();
                updateCountArray(currWord);

                countNode *currNode = countHd;
                int found = 0;
                while (currNode != NULL){
                        if (checkCounts(currNode)){
                                currNode->strTail->groupNext = newStr;
                                currNode->strTail = newStr;
                                found = 1;
                                break;
                        }

                        currNode = currNode->nextCount;
                }

                if (!found){
                        countNode *newCount = calloc(1,sizeof(countNode));
                        if(newCount == NULL){
                                fprintf(stderr,"Memory Allocation Failure.\n");
                                return 1;
                        }
                        updateCountNode(newCount);
                        newCount->strHd = newCount->strTail = newStr;
                        newCount->nextCount = NULL;
                        countTail->nextCount = newCount;
                        countTail = newCount;
                }

        }

        printList();

        return errSeen;
}

int badWord(){
        char *currChar = currWord;
        while (*currChar != '\0'){
                if(!isalpha(*currChar) && !isspace(currChar)){
                        return 1;
                }
                currChar++;
        }
        return 0;
}
void initCountArray(){
        int i;
        for (i = 0;i < 26; i++){
                countArray[i] = 0;
        }
}

void updateCountArray(char *word){
        char *currChar = word;
        while (*currChar != '\0'){
                if(isalpha(*currChar)){
                        countArray[tolower(*currChar) - 'a']++;
                }
                currChar++;
        }
}

void updateCountNode(countNode *node){
        int i;
        for(i = 0;i < 26; i++){
                node->count[i] = countArray[i];
        }
}

int checkCounts(countNode *currNode){
        int i;
        for(i = 0; i < 26; i++){
                if(countArray[i] != currNode->count[i]){
                        return 0;
                }
        }
        return 1;
}

void printList(){
        countNode *currCntNode = countHd;
        while (currCntNode != NULL){
                strNode *currStrNode = currCntNode->strHd;
                while (currStrNode != NULL){
                        printf("%s ",currStrNode->word);
                        currStrNode = currStrNode->groupNext;
                }
                printf("%c",'\n');
                currCntNode = currCntNode->nextCount;
        }
}
